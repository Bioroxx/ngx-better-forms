formGroup = this.formBuilder.group(
  {
    field1: new FormControl<string>(''),
    target: new FormControl<string>(''),
  },
  {
    validators: [
      BetterValidation.conditionalValidators({
        targetControlPath: 'target',
        targetValidators: [Validators.required],
        conditions: [
          {
            controlPath: 'field1',
            testValues: ['a'],
          },
        ],
        options: {
          addValidatorsCallback: () =>
            alert('Validator added to target field!'),
          removeValidatorsCallback: () =>
            alert('Validator removed from target field!'),
        },
      }),
    ],
  },
);
